311 QString Jpeg2000Transcoder::encodeUrl(QString stringToEncode)
312 {
313  QString encodedString = "";
314  //convert our Utf8 QString to Local 8 bit standard string
315  std::string stdStringToEncode = stringToEncode.toLocal8Bit().constData();
316  for(int i = 0; i < stringToEncode.length(); i++)
317  {
318  if((stringToEncode.at(i) >= '0' && stringToEncode.at(i) <= '9')
319  || (stringToEncode.at(i) >= 'A' && stringToEncode.at(i) <= 'Z')
320  || (stringToEncode.at(i) >= 'a' && stringToEncode.at(i) <= 'z')
321  || stringToEncode.at(i) == '.'
322  || stringToEncode.at(i) == ':'
323  || stringToEncode.at(i) == '/'
324  || stringToEncode.at(i) == '?'
325  || stringToEncode.at(i) == '\0')
326  {
327  encodedString.append(stringToEncode.at(i));
328  }
329  else
330  {
331  //get ordinal value, convert it to string as hexadecimal and append it with % to our result string
332  //we have to use standard string's [] operator here because QChar is Utf8
333  encodedString.append("%" + QString::number((unsigned char)stdStringToEncode[i],16));
334  }
335  }
336  return encodedString;
337 }
